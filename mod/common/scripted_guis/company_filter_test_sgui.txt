com_test_set_variable = {
    scope = player
    
    effect = {
        set_variable = {
            name = com_test_var
            value = 100
        }
    }
}

com_test_clear_variable = {
    scope = player
    
    effect = {
        if = {
            limit = { has_variable = com_test_var }
            remove_variable = com_test_var
        }
    }
}
