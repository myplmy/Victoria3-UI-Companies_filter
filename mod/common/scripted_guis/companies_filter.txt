# 파일 위치: mod/common/scripted_guis/companies_filter.txt
# Company 필터링을 위한 ScriptedGui 정의

# 필터 초기화
clear_company_filter = {
	scope = country
	
	is_shown = {
		always = yes
	}
	
	is_valid = {
		always = yes
	}
	
	effect = {
		# 필터 변수 제거
		remove_variable = filter_building_type
		
		# 모든 필터 플래그 제거
		remove_variable = filter_wheat_farm
		remove_variable = filter_coal_mine
		remove_variable = filter_textile_mills
		remove_variable = filter_steel_mills
		remove_variable = filter_oil_rig
	}
}

# Wheat Farm 필터
filter_by_wheat_farm = {
	scope = country
	
	is_shown = {
		always = yes
	}
	
	is_valid = {
		always = yes
	}
	
	effect = {
		# 다른 필터 제거
		remove_variable = filter_coal_mine
		remove_variable = filter_textile_mills
		remove_variable = filter_steel_mills
		remove_variable = filter_oil_rig
		
		# 현재 필터 설정
		set_variable = {
			name = filter_building_type
			value = flag:wheat_farm
		}
		set_variable = filter_wheat_farm
	}
}
